Соединение с Базой данных 

$dbc = mysql_connect('Хост размещения БД', 'Имя пользователя БД', 'Пароль пользователя БД', 'Имя БД')
or die ('Ошибка соединения с MySQL-сервером');
Пример:
    $dbc = mysql_connect('localhost', 'root', 'root', 'my_bd')
    or die ('Ошибка соединения с MySQL-сервером');

Составление строки запроса

$query = "INSERT INTO имя таблицы (Имена, калонок, таблицы)" .
"VALUES ('Данные', 'добавляемые ', 'в', 'таблицу')"; 
Пример: 
    $query = "INSERT INTO table_name(first_name, last_name)" . 
    "VALUES ('$first_name', '$last_name')";
//Имена колонок строго совпадают с добавляемыми данными

Выполнение запроса

$result = mysqli_query(Ссылка на соединение с БД $bdc, запрос $query)
or die ('Ошибка выполнения запроса');

Закрытие соединения

mysqli_close(Ссылка на соединение с БД $bdc);
